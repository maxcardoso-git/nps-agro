**Networks:**
- app-net: internal network for all services (bridge driver)
- db-net: external network for connecting to the existing PostgreSQL container on the VPS
  ```yaml
  networks:
    app-net:
      driver: bridge
    db-net:
      external: true
      name: ${DB_NETWORK:-nps-db-net}
  ```
  The DB_NETWORK env var allows configuring the actual Docker network name on the VPS where PostgreSQL runs. Default to "nps-db-net". On the VPS, the operator needs to ensure this network exists and the PostgreSQL container is attached to it.

Create `nginx/` directory in project root.

**Create `nginx/nginx.conf`:**

```nginx
worker_processes auto;

events {
    worker_connections 1024;
}

http {
    include       /etc/nginx/mime.types;
    default_type  application/octet-stream;

    sendfile on;
    keepalive_timeout 65;

    # Gzip compression
    gzip on;
    gzip_types text/plain text/css application/json application/javascript text/xml;

    # Upstreams (service names from docker-compose)
    upstream backend {
        server nps-backend:3310;
    }

    upstream user_portal {
        server nps-user-portal:3000;
    }

    upstream admin_console {
        server nps-admin-console:3000;
    }

    upstream analytics {
        server nps-analytics:3000;
    }

    server {
        listen 80;
        server_name _;

        # API requests -> backend
        location /api/ {
            proxy_pass http://backend/;
            proxy_set_header Host $host;
            proxy_set_header X-Real-IP $remote_addr;
            proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
            proxy_set_header X-Forwarded-Proto $scheme;
        }

        # Health check -> backend
        location /health {
            proxy_pass http://backend/health;
            proxy_set_header Host $host;
        }

        # Admin console -> /admin path
        location /admin/ {
            proxy_pass http://admin_console/;
            proxy_set_header Host $host;
            proxy_set_header X-Real-IP $remote_addr;
            proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
            proxy_set_header X-Forwarded-Proto $scheme;
        }

        # Analytics -> /analytics path
        location /analytics/ {
            proxy_pass http://analytics/;
            proxy_set_header Host $host;
            proxy_set_header X-Real-IP $remote_addr;
            proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
            proxy_set_header X-Forwarded-Proto $scheme;
        }

        # Default -> user portal (main site)
        location / {
            proxy_pass http://user_portal;
            proxy_set_header Host $host;
            proxy_set_header X-Real-IP $remote_addr;
            proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
            proxy_set_header X-Forwarded-Proto $scheme;
        }
    }
}
```

Routing logic:
- `/api/*` -> backend (strips /api/ prefix, so /api/health becomes /health on backend)
- `/health` -> backend health check directly
- `/admin/*` -> admin-console frontend
- `/analytics/*` -> analytics frontend
- `/*` (everything else) -> user-portal (default, main site)

Note: The backend NestJS routes do NOT have an /api prefix in the code. The Nginx `/api/` location strips the prefix via `proxy_pass http://backend/` (trailing slash strips the matched prefix). This means frontend clients should call `/api/health`, `/api/interviews`, etc. through Nginx.

IMPORTANT: If the frontends currently call the backend directly via NEXT_PUBLIC_API_URL with the full host:port (e.g., http://72.61.52.70:3310), that still works because the backend port 3310 is also exposed directly. The Nginx /api/ route is an additional convenience for browser-side requests. The NEXT_PUBLIC_API_URL/NEXT_PUBLIC_API_BASE_URL build args should point to the direct backend URL (http://72.61.52.70:3310) for server-side rendering, or to the Nginx URL (http://72.61.52.70/api) for client-side requests. Keep using the direct URL for now since that is what the frontends expect.
